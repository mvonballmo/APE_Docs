Deklarieren: var

          
function varTest() {
  var x = 1;
  if (true) {
    var x = 2;  // gleiche Variable!
    console.log(x);  // 2
  }
  console.log(x);  // 2
}

Deklarieren: let

          
function letTest() {
  let x = 1;
  if (true) {
    let x = 2;  // andere Variable
    console.log(x);  // 2
  }
  console.log(x);  // 1
}

Deklarieren: const

          
function constTest() {
  const x = 1;
  if (true) {
    const x = 2;  // andere Variable
    console.log(x);  // 2
  }
  x = 2;  // Nicht erlaubt!
  console.log(x);  // 1
}

Sets

Speichert eindeutige Werte jedes beliebigen Typs.

instanceof

Prüft, ob ein Objekt durch eine bestimmte Constructor-Funktion erstellt wurde.

          
            function person(firstName, lastName) {
              this.firstName = firstName;
              this.lastName = lastName;
            }
            const bob = new person('John', 'Doe');
            const isPerson = bob instanceof person; // true
            const isObject = bob instanceof Object; // true
            const isNumber = bob instanceof Number; // false

instanceof (Klassen)

Prüft, ob ein Objekt von einer bestimmten Klasse abgeleitet ist.

          
            class person {
              constructor(name) {
                this.name = name;
              }
            }
            class employee extends person {
              constructor(name) { super(name); }
            }

            const bob = new employee('John', 'Doe');
            const isPerson = bob instanceof employee; // true
            const isPerson = bob instanceof person;   // true
            const isObject = bob instanceof Object;   // true
            const isNumber = bob instanceof Number;   // false

Rechenoperatoren

  • +, -, *, /, %, **
  • % berechnet den Restwert einer ganzzahligen Division (Modulo): 5 % 2 == 1.
  • 5 ** 2 ist 5 hoch 2 = 25
  • ++ ist wie +1; -- ist wie -1
  • ++zahl erhöht zahl vor der Auswertung um 1.
  • zahl++ erhöht zahl nach der Auswertung um 1.
  • Negative Zahlen werden mit - geschrieben -zahl.
  • Es gilt Punkt vor Strich und dann links vor rechts.
  • Um die Reihenfolge zu ändern können Klammern () gesetzt werden.

Vergleichsoperatoren

  • ==, !=, >, <, <=, >=, ===, !==
          
            const x = 1 == '1';   // x ist 'true'
            const y = 1 === '1';  // y ist 'false'

Boolesche Operatoren

  • &&, ||, !
          
            const x = false;
            const y = true;
            const a = x && y; // a ist 'false'
            const o = x || y; // o ist 'true'
            const n = !x;     // n ist 'true'

Nullesche Operatoren

  • ?., ??

Spread (...) Operator

Kopiert alle Eigenschaften eines Objekts.

          
            const person = {
              a: 1,
              b: "n",
              c: false,
              d: {
                e: 2,
                f: "m"
              }
            };
            const person2 = person;                         // Referenz
            const personCopy1 = Object.assign({}, person);  // Kopie
            const personCopy2 = { ...person };              // Kopie
            person.a = 2;  // Ändert nur person und person2

Spread mit Anpassungen

Kann mit explizitem Setzen von Eigenschaften kombiniert werden.

          
            const person = { a: 1, b: "n", c: false, d: { e: 2, f: "m" } };
            const person2 = {
              ...person,
              a: 2,
              d: {
                ...person.d,
                e: 1
              }
            } // { a: 2, b: "n", c: false, d:  { e: 1, f: "m" } }

Spread mit Arrays

Kopiert nicht nur Objekte sondern auch Arrays.

          
            const a = [0, 1, 2];
            const b = [3, 4, 5];

            const b2 = [...b];            // Kopie von b

            const c = [...a, ...b];       // [0, 1, 2, 3, 4, 5, 6]
            const d = [0, 1, 2, ...b, 7]; // [0, 1, 2, 3, 4, 5, 6, 7]

Spread mit Parametern

Kann auch als Funktionsparametern benutzt werden.

          
            function sum(x, y, z) {
              return x + y + z;
            }
            const numbers = [1, 2, 3];
            const total = sum(...numbers);  // Ergibt 6

Destructuring

Mit [] können Daten aus Arrays extrahiert werden.

          
            const items = [10, 20, 30, 40, 50];
            const [a, b, ...rest] = items;
            // a == 10, b == 20, rest == [30, 40, 50]

            // Mit Standardwerte
            const [c, d, e = -1, f = -1] = rest;
            // c == 30, d == 40, e = 50, f = -1

Destructuring

Mit {} können Daten aus Objekten extrahiert werden.

          
            const obj = { name: "bob", age: 10, country: "CH" };
            const { name, age } = obj;
            // name == "bob", age == 10

            let name2, age2;          // Semicolon nötig
            ({ name2, age2 } = obj);  // Klammern nötig
            // name2  = "bob", age2 = 10

Destructuring

Daten aus Objekten zu anderen Eigenschaften zuweisen.

          
            const obj = { name: "bob", age: 10, country: "CH" };
            const { name: firstName, age: years } = obj;
            // firstName == "bob", years == 10

            // Mit Standardwerte
            const {name: firstName = "John", lastName = "Doe"} = obj;
            // firstName  = "bob", lastName = "Doe"

            // Mit "rest"
            const {name, ...obj2} = obj;
            // name = "bob", obj2 = { age: 10, country: "CH" }

Verzweigungen: if/else 

          
            const x = 42;
            let y;
            if (x > 42) {
              y = 1;
            } else {
              y = 2;
            }
            // y == 2

Ternäre Operator (?:) 

          
            const x = 42;
            const y = x > 42 ? 1 : 2;
            // y == 2

Verzweigungen: switch 

          
            const x = 42;
            let y;
            switch (x) {
              case 41:
                y = 1;
                break;
              case 42:
                y = 2;
                break;
              default:
                y = 3;
            }
            // y == 2

Schleifen: for

Für sich wiederholende Programmteile werden Schleifen verwendet.

          
            for (let i = 0; i < 5; i++) {
              // Wird für i = 0,1,2,3,4 ausgeführt
            }

Schleifen: for..in

Um allen Eigenschaften eines Objekts durchzugehen.

          
            const o = { a: 1, b: 2, c: 3 };

            for (const property in o) {
              // Ergibt "a", "b", "c"
            }

Schleifen: for..of

Um allen Elementen eines Arrays durchzugehen.

          
            const a = ['a', 'b', 'c'];

            for (const element of array1) {
              // Ergibt 'a', 'b', 'c'
            }

Schleifen: while

Um Code auszuführen bis eine Bedingung erfüllt wird.

          
            let n = 0;

            while (n < 3) {
              n++;
            }
            // n ist 3

Schleifen: do..while

Um Code mindestens ein mal auszuführen und dann bis eine Bedingung erfüllt wird.

          
            let result = '';
            let i = 0;

            do {
              i = i + 1;
              result = result + i;
            } while (i < 5);
            // result ist "12345"

Exception-Handling

try wird bis einen Fehler vorkommt ausgeführt
catch(error) wird nur bei einem Fehler im try {} ausgeführt
finally wird immer ausgeführt

Exception-Handling

  • throw erzeugt ein Exception
  • Aufrufe werden bis zu einem catch "aufgerollt"
  • Mangels ein catch wird die Applikation beendet
          
            const divide = (a, b) => {
              if (b === 0) { throw "Divide by zero"; }

              return a / b;
            };

            try {
              divide(1, 0);
            }
            catch (error) {
              console.log(error);
            }

Gewöhnliche Funktionen

  • Kann mehrere Parameter haben
  • "rest" Parameter wird mit "..." geschrieben
  • Parametern können default-Werte haben
          
            function add(a, b, c = 0) {
              return a + b + c;
            }
            function addAll(...args) {
              let result = 0;
              for (const a of args) {
                result += a;
              }
              return result;
            }
            const sum1 = add(1, 2);           // 3
            const sum2 = add(1, 2, 3);        // 6
            const sum3 = add(1, 2, 3, 4);     // 6 weil "4" ignoriert wird
            const sum4 = addAll(1, 2, 3, 4);  // 10
            const sum5 = addAll(...[1,2,3]);  // 6

Methoden

Funktionen, die an einem Objekt gebunden sind.

          
            const person = {
              age: 34,

              setAge(a) {
                this.age = a;
              }

              getYearsLeft() {
                return 65 - this.age;
              }
            };

            person.setAge(42);
            const y = person.getYearsLeft();  // 23

Getters/Setters

Ermöglicht berechnete Eigenschaften.

          
            const person = {
              _age: 34;

              get age() {
                return _age;
              }

              set age(a) {
                this._age = a - 10; // Don't lie about your age 😉...
              }
            };

            person.age = 42;
            const y = person.age;  // 32

Arrow Funktionen

  • Eine kompaktere Definitionsform für Funktionen.
  • Für Methoden nicht gut geeignet (weil this eine andere Bedeutung hat).
          
            const add = (a, b, c = 0) => a + b + c;
            const addAll = (...args) => {
              let result = 0;
              for (const a of args) {
                result += a;
              }
              return result;
            }
            const singleArgument = a => a + 2;  // Benötigt keine Klammern
            const noParameters = () => false;   // Benötigt leere Klammern
